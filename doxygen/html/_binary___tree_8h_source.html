131template<typename Item_Type>
132 Binary_Tree<Item_Type> Binary_Tree<Item_Type>::
133 read_binary_tree(std::istream& in) {
134 std::string next_line;
135 getline(in, next_line);
136 if (next_line == "NULL") {
137 return Binary_Tree<Item_Type>();
138 } else {
139 Item_Type the_data;
140 std::istringstream ins(next_line);
141 ins >> the_data;
142 Binary_Tree<Item_Type> left = read_binary_tree(in);
143 Binary_Tree<Item_Type> right = read_binary_tree(in);
144 return Binary_Tree<Item_Type>(the_data, left, right);
145 }
